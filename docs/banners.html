from banners.slides import Cover, Intro, Section, Closing
from banners.content import Text, Image, Graph
from banners.palette import Palette, BLUE, GREEN

Cover(
    title="My Project",
    subtitle="Context in one line.",
    date="April 2026",
    content=[
        Text("**Point 1** — Something relevant."),
        Text("**Point 2** — Concrete result."),
        Text("**Point 3** — Team impact."),
    ],
    content_kind="neutral",
).render()

 25class Slide:
 26    """Abstract base class for all slide types.
 27
 28    Subclasses must implement `_render_banner` to produce the banner HTML.
 29    The `render` method assembles the full slide as a `mo.vstack`.
 30
 31    Content layout is resolved automatically based on what is passed:
 32
 33    - **Single item** — rendered at full width.
 34    - **List of `Text` only** — rendered as equal columns via `mo.hstack`.
 35      Pass `content_kind` to wrap each column in a `mo.callout` box.
 36    - **List with any `Image` or `Graph`** — rendered as an equal-width CSS
 37      grid (`1fr` per column) automatically, no extra parameter needed.
 38
 39    Args:
 40        title: Main heading of the slide.
 41        subtitle: Support phrase displayed inside the banner.
 42        content: Content elements below the banner. Accepts a single
 43            `Text`, `Image`, or `Graph` instance, a list of them, or any
 44            marimo component.
 45        palette: Color palette for the banner. Defaults to `ORANGE`.
 46        content_kind: Box style applied to `Text` items when the content is
 47            a list of text-only elements. Accepted values: `"neutral"`,
 48            `"info"`, `"success"`, `"warn"`, `"danger"`.
 49

 51        ```python

 93    def _render_banner(self) -> mo.Html:
 94        """Produce the banner HTML for this slide type.
 95
 96        Raises:
 97            NotImplementedError: Must be implemented by each subclass.
 98        """
 99        raise NotImplementedError(f"{type(self).__name__} must implement _render_banner()")

101    def _render_content(self) -> list:
102        """Resolve content items into marimo components and apply layout.
103
104        Returns:
105            A list of marimo components to be stacked below the banner.
106        """
107        if self.content is None:
108            return []
109
110        items = self.content if isinstance(self.content, list) else [self.content]
111
112        if len(items) == 1:
113            return [self._render_item(items[0])]
114
115        from .content.text import Text as _Text
116        has_non_text = any(not isinstance(i, (_Text, str)) for i in items)
117
118        rendered = [
119            self._render_item(item, wrap_kind=self.content_kind)
120            for item in items
121        ]
122
123        if has_non_text:
124            return [self._css_grid(rendered)]
125        return [mo.hstack(rendered, justify="space-between")]
126
127    def _render_item(self, item: ContentItem, wrap_kind: "str | None" = None):
128        """Render a single content item into a marimo component.
129
130        Args:
131            item: A `Text`, `Image`, `Graph`, plain string, or marimo component.
132            wrap_kind: If set and the item is `Text` or `str`, wraps the
133                rendered markdown in a `mo.callout` with this kind.
134
135        Returns:
136            A marimo component.
137        """
138        from .content.text import Text as _Text
139
140        if isinstance(item, (_Text, str)):
141            md = mo.md(item.body if isinstance(item, _Text) else item)
142            return mo.callout(md, kind=wrap_kind) if wrap_kind else md
143
144        if hasattr(item, "render") and callable(item.render):
145            return item.render()
146
147        return item

149    @staticmethod
150    def _css_grid(items: list) -> mo.Html:
151        """Wrap rendered items in an equal-width CSS grid container.
152
153        Args:
154            items: List of marimo `mo.Html` components.
155
156        Returns:
157            A `mo.Html` component with a CSS grid layout (`1fr` per column).
158        """
159        cols = " ".join(["1fr"] * len(items))
160        parts = []
161        for item in items:
162            html = item.text if hasattr(item, "text") else str(item)
163            parts.append(f'<div style="min-width:0;">{html}</div>')
164        inner = "".join(parts)
165        return mo.Html(
166            f'<div style="display:grid;grid-template-columns:{cols};'
167            f'gap:1rem;align-items:start;">{inner}</div>'
168        )

170    @staticmethod

172        return (
173            f'<div style="font-size:0.85rem;letter-spacing:0.2em;text-transform:uppercase;'
174            f'color:{color};margin-bottom:0.5rem;">{text}</div>'

177    @staticmethod
178    def _icon_src(icon) -> str:
179        """Convert a path, URL, or bytes to a base64 data URI.
180
181        Args:
182            icon: File path (`str` or `Path`), URL string, or raw `bytes`.
183
184        Returns:
185            A `data:image/<mime>;base64,...` string.
186        """
187        _EXT_MIME = {
188            "jpg": "jpeg", "jpeg": "jpeg", "png": "png",
189            "svg": "svg+xml", "gif": "gif", "webp": "webp",
190        }
191        if isinstance(icon, str) and icon.startswith(("http://", "https://")):
192            ext = icon.split("?")[0].rsplit(".", 1)[-1].lower()
193            mime = _EXT_MIME.get(ext, "png")
194            req = urllib.request.Request(icon, headers={"User-Agent": "Mozilla/5.0"})
195            ctx = ssl.create_default_context()
196            ctx.check_hostname = False
197            ctx.verify_mode = ssl.CERT_NONE
198            with urllib.request.urlopen(req, timeout=10, context=ctx) as r:
199                data = r.read()
200                ct = r.headers.get("Content-Type", "")
201            if "svg" in ct:
202                mime = "svg+xml"
203            elif "jpeg" in ct or "jpg" in ct:
204                mime = "jpeg"
205            elif "png" in ct:
206                mime = "png"
207            b64 = base64.b64encode(data).decode()
208            return f"data:image/{mime};base64,{b64}"
209
210        if isinstance(icon, (str, Path)):
211            p = Path(icon)
212            mime = _EXT_MIME.get(p.suffix.lower().lstrip("."), "png")
213            b64 = base64.b64encode(p.read_bytes()).decode()
214            return f"data:image/{mime};base64,{b64}"
215
216        b64 = base64.b64encode(icon).decode()
217        return f"data:image/png;base64,{b64}"

219    def _icon_corner_html(self, icon: "dict | None", bottom: str, right: str, default_size: str) -> str:
220        if icon is None:
221            return ""
222        try:
223            size = icon.get("size", default_size)
224            src = self._icon_src(icon["src"])
225            return (
226                f'<img src="{src}" '
227                f'style="position:absolute;bottom:{bottom};right:{right};'
228                f'height:{size};opacity:0.85;border-radius:4px;">'
229            )
230        except Exception:
231            return ""

233    def _icon_inline_html(self, icon: "dict | None", default_size: str, style: str = "") -> str:
234        if icon is None:
235            return ""

238            src = self._icon_src(icon["src"])
239            return f'<img src="{src}" style="height:{size};{style}border-radius:4px;">'
240        except Exception:
241            return ""

Palette for Section slides: left border accent over a dark background.

Attributes:
  • border: Left-border accent color (hex).
  • bg_start: Dark background gradient start color (hex).
  • bg_end: Dark background gradient end color (hex).
  • text_main: Primary text color.
  • text_sub: Subtitle text color.
